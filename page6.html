Ví dụ ứng dụng
Prompt: Viết một hàm Python để tìm số Fibonacci thứ n sử dụng đệ quy có nhớ 
def fibonacci(n, memo={}):
    if n in memo:
        return memo[n]
    if n <= 2:
        return 1
    memo[n] = fibonacci(n-1, memo) + fibonacci(n-2, memo)
    return memo[n]

# Sử dụng hàm
print(fibonacci(10))  # Output: 55
